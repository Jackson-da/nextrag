## 1. 项目概述

### 1.1 项目目标

构建一个**企业级智能文档问答系统**，让用户可以：

1. **上传文档**（PDF、Word、TXT 等）
2. **自动解析**文档内容并分块
3. **构建知识库**，自动向量化存储
4. **智能问答**，基于文档内容回答问题
5. **多轮对话**，支持追问和上下文理解

### 1.2 功能演示

```
用户上传文档
    ↓
系统解析 → 分块 → 向量化 → 存储到向量数据库
    ↓
用户提问："文档主要讲了什么？"
    ↓
系统检索相关片段 → 构建提示词 → 调用 LLM → 返回答案
```

### 1.3 技术栈

| 层级 | 技术 | 作用 |
|-----|------|-----|
| **前端** | Vue3 + Element Plus | 用户界面 |
| **后端** | FastAPI | API 服务 |
| **核心框架** | LangChain / LangChain-Core | RAG 实现 |
| **向量库** | Chroma | 向量存储和检索 |
| **Embedding** | BGE / Jina | 文本向量化 |
| **LLM** | DeepSeek-chat / Qwen | 答案生成 |
| **文档解析** | PyPDF2 / python-docx | 文档读取 |
| **数据库** | SQLite | 元数据存储 |

## 5. 文档处理模块

### 5.1 文档加载器

```python
# backend/app/core/document_loader.py
"""
文档加载模块
支持多种格式文档的读取和解析
"""

from langchain_community.document_loaders import (
    PyPDFLoader,      # PDF 文件加载
    Docx2txtLoader,    # Word 文件加载
    TextLoader,        # 文本文件加载
)
from langchain_core.documents import Document
from pathlib import Path
from typing import List, Union
import logging

# 配置日志
logger = logging.getLogger(__name__)


class DocumentLoader:
    """
    文档加载器类
    
    功能：
    - 根据文件扩展名自动选择加载器
    - 统一接口加载所有支持的文档格式
    
    使用方式：
    loader = DocumentLoader()
    docs = loader.load("path/to/document.pdf")
    """
    
    # 扩展名到加载器的映射
    LOADER_MAPPING = {
        ".pdf": PyPDFLoader,      # PDF 文件
        ".docx": Docx2txtLoader,  # Word 文件
        ".doc": Docx2txtLoader,   # 旧版 Word
        ".txt": TextLoader,       # 纯文本
        ".md": TextLoader,        # Markdown
    }
    
    def __init__(self, encoding: str = "utf-8"):
        """
        初始化加载器
        
        Args:
            encoding: 文件编码，默认 utf-8
        """
        self.encoding = encoding
    
    def load(self, file_path: str) -> List[Document]:
        """
        加载文档
        
        Args:
            file_path: 文件路径
            
        Returns:
            Document 列表，每个 Document 包含：
            - page_content: 文本内容
            - metadata: 元数据（来源、页码等）
            
        Raises:
            ValueError: 不支持的文件类型
            FileNotFoundError: 文件不存在
        """
        # 检查文件是否存在
        path = Path(file_path)
        if not path.exists():
            raise FileNotFoundError(f"文件不存在: {file_path}")
        
        # 获取扩展名（小写）
        ext = path.suffix.lower()
        
        # 检查是否支持该类型
        if ext not in self.LOADER_MAPPING:
            supported = ", ".join(self.LOADER_MAPPING.keys())
            raise ValueError(f"不支持的文件类型: {ext}，支持的类型: {supported}")
        
        # 获取对应的加载器类
        loader_class = self.LOADER_MAPPING[ext]
        
        # 创建加载器实例并加载
        logger.info(f"正在加载文件: {file_path}")
        
        try:
            # 不同加载器的初始化方式可能不同
            if ext == ".txt" or ext == ".md":
                # 文本文件需要指定编码
                loader = loader_class(file_path, encoding=self.encoding)
            else:
                loader = loader_class(file_path)
            
            # 加载文档
            docs = loader.load()
            
            # 添加文件名校验到元数据
            for doc in docs:
                doc.metadata["source"] = path.name
                doc.metadata["file_path"] = str(path)
            
            logger.info(f"成功加载 {len(docs)} 个文档片段")
            return docs
            
        except Exception as e:
            logger.error(f"加载文件失败: {e}")
            raise
    
    def load_batch(self, file_paths: List[str]) -> List[Document]:
        """
        批量加载多个文档
        
        Args:
            file_paths: 文件路径列表
            
        Returns:
            所有文档的 Document 列表
        """
        all_docs = []
        
        for path in file_paths:
            try:
                docs = self.load(path)
                all_docs.extend(docs)
            except Exception as e:
                logger.warning(f"跳过文件 {path}: {e}")
                continue
        
        return all_docs


# 便捷函数
def load_document(file_path: str) -> List[Document]:
    """
    便捷函数：加载单个文档
    
    使用方式：
    docs = load_document("path/to/file.pdf")
    """
    loader = DocumentLoader()
    return loader.load(file_path)
```

### 5.2 文本分割器

```python
# backend/app/core/text_splitter.py
"""
文本分割模块
将长文档分割成适合检索的小块
"""

from langchain_text_splitters import (
    RecursiveCharacterTextSplitter,  # 推荐：智能分割
    CharacterTextSplitter,           # 简单按字符分割
    MarkdownTextSplitter,              # Markdown 专用
)
from langchain_core.documents import Document
from typing import List, Optional
import logging

logger = logging.getLogger(__name__)


class TextSplitter:
    """
    文本分割器
    
    为什么需要分割？
    - LLM 有上下文长度限制
    - 检索时小块更容易找到精确信息
    - 减少 token 消耗，降低成本
    
    分割策略：
    1. 按段落分割（保留语义）
    2. 按句子分割（保留完整性）
    3. 固定长度分割（简单粗暴）
    """
    
    def __init__(
        self,
        chunk_size: int = 500,
        chunk_overlap: int = 50,
        length_function: callable = len,
    ):
        """
        初始化分割器
        
        Args:
            chunk_size: 每段最大字符数
            chunk_overlap: 段之间重叠的字符数
                         重叠可以避免关键信息被切断
            length_function: 计算文本长度的函数
        """
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.length_function = length_function
        
        # 创建分割器实例
        self.splitter = RecursiveCharacterTextSplitter(
            separators=[
                "\n\n",      # 优先按段落分割（保留完整语义）
                "\n",        # 其次按行分割
                "。",         # 中文句号
                "！",         # 中文感叹号
                "？",         # 中文问号
                "；",         # 中文分号
                "，",         # 中文逗号
                " ",         # 英文空格
                "",
            ],
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            length_function=length_function,
        )
    
    def split_documents(
        self,
        documents: List[Document],
        metadata: Optional[dict] = None
    ) -> List[Document]:
        """
        分割文档列表
        
        Args:
            documents: Document 对象列表
            metadata: 附加元数据（会合并到每个文档）
            
        Returns:
            分割后的 Document 列表
        """
        if not documents:
            return []
        
        logger.info(f"开始分割 {len(documents)} 个文档")
        
        # 执行分割
        chunks = self.splitter.split_documents(documents)
        
        # 添加附加元数据
        if metadata:
            for chunk in chunks:
                chunk.metadata.update(metadata)
        
        logger.info(f"分割完成，得到 {len(chunks)} 个文本块")
        
        return chunks
    
    def split_text(
        self,
        text: str,
        metadata: Optional[dict] = None
    ) -> List[Document]:
        """
        分割单个文本
        
        Args:
            text: 文本内容
            metadata: 附加元数据
            
        Returns:
            Document 列表
        """
        # 先创建 Document 对象
        doc = Document(page_content=text, metadata=metadata or {})
        
        # 再分割
        return self.split_documents([doc])


class MarkdownSplitter:
    """
    Markdown 专用分割器
    
    特点：
    - 按标题层级分割
    - 保留 Markdown 格式结构
    - 适合技术文档
    """
    
    def __init__(
        self,
        chunk_size: int = 500,
        chunk_overlap: int = 50,
    ):
        self.splitter = MarkdownTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
        )
    
    def split_documents(self, documents: List[Document]) -> List[Document]:
        """按 Markdown 结构分割文档"""
        return self.splitter.split_documents(documents)


# 配置工厂函数
def create_splitter(
    splitter_type: str = "recursive",
    chunk_size: int = 500,
    chunk_overlap: int = 50,
) -> TextSplitter:
    """
    创建分割器
    
    Args:
        splitter_type: 分割器类型
                     - "recursive": 递归智能分割（推荐）
                     - "character": 按字符分割
                     - "markdown": Markdown 专用
        chunk_size: 每块最大字符数
        chunk_overlap: 重叠字符数
    """
    if splitter_type == "recursive":
        return TextSplitter(chunk_size=chunk_size, chunk_overlap=chunk_overlap)
    elif splitter_type == "markdown":
        return MarkdownSplitter(chunk_size=chunk_size, chunk_overlap=chunk_overlap)
    else:
        # 默认使用递归分割器
        return TextSplitter(chunk_size=chunk_size, chunk_overlap=chunk_overlap)
```

```python
# backend/app/core/vectorstore.py
"""
向量存储模块
实现文档的向量化存储和检索
"""

from langchain_deepseek import ChatDeepSeek
from langchain_chroma import Chroma
from langchain_core.documents import Document
from langchain_core.embeddings import Embeddings
from typing import List, Optional, Tuple
import os
import logging

logger = logging.getLogger(__name__)


class VectorStoreManager:
    """
    向量存储管理器
    
    功能：
    1. 创建/加载向量数据库
    2. 添加文档到向量库
    3. 相似性检索
    4. 删除文档
    
    使用 Chroma 作为向量数据库的原因：
    - 轻量级，易于部署
    - 支持持久化
    - LangChain 官方集成
    - 适合中小规模数据
    """
    
    def __init__(
        self,
        persist_directory: str,
        embedding_model: Optional[Embeddings] = None,
        collection_name: str = "documents",
    ):
        """
        初始化向量存储管理器
        
        Args:
            persist_directory: 持久化存储目录
            embedding_model: Embedding 模型（不传则创建默认）
            collection_name: 集合名称
        """
        self.persist_directory = persist_directory
        
        # 确保目录存在
        os.makedirs(persist_directory, exist_ok=True)
        
        # 初始化 Embedding 模型（使用 BGE 中文优化模型）
        if embedding_model is None:
            from langchain_community.embeddings import HuggingFaceBgeEmbeddings
            self.embedding_model = HuggingFaceBgeEmbeddings(
                model_name="BAAI/bge-large-zh-v1.5",
                model_kwargs={'device': 'cpu'},
                encode_kwargs={'normalize_embeddings': True}
            )
        else:
            self.embedding_model = embedding_model
        
        self.collection_name = collection_name
        
        # 初始化向量数据库
        self._init_vectorstore()
    
    def _init_vectorstore(self):
        """
        初始化或加载向量数据库
        
        Chroma 会自动：
        - 如果目录存在，加载已有数据
        - 如果目录不存在，创建新的
        """
        self.vectorstore = Chroma(
            collection_name=self.collection_name,
            embedding_function=self.embedding_model,
```

### 6.2 RAG 链构建

```python
# backend/app/core/rag_chain.py
"""
RAG 链构建模块
实现检索增强生成的完整流程
"""

from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough
from langchain_deepseek import ChatDeepSeek
from langchain.chains import create_retrieval_chain
from langchain.chains.combine_documents import create_stuff_documents_chain
from typing import Optional, List, Dict, Any
import logging

logger = logging.getLogger(__name__)


class RAGChainBuilder:
    """
    RAG 链构建器
    
    负责：
    1. 构建检索链
    2. 配置提示词
    3. 处理检索结果
    
    RAG 链的工作原理：
    用户问题 → 检索相关文档 → 构建提示词 → LLM 生成答案
    """
    
    # 默认系统提示词
    DEFAULT_SYSTEM_PROMPT = """你是一个专业的文档问答助手。

你的任务是：
1. 基于提供的上下文（context）回答用户问题
2. 如果上下文中没有相关信息，请明确告知用户
3. 回答要准确、清晰、简洁
4. 可以适当引用上下文中的原文

请记住：
- 只使用上下文中的信息回答，不要编造
- 如果不确定，说明不确定的原因
- 回答尽量使用中文

上下文：
{context}
"""
    
    def __init__(
        self,
        retriever,
        llm_model: str = "deepseek-chat",
        temperature: float = 0.0,
        system_prompt: Optional[str] = None,
    ):
        """
        初始化 RAG 链构建器
        
        Args:
            retriever: 检索器（from vectorstore.as_retriever()）
            llm_model: LLM 模型名称
            temperature: 生成温度（0=确定性，1=创造性）
            system_prompt: 自定义系统提示词
        """
        self.retriever = retriever
        self.llm_model = llm_model
        self.temperature = temperature
        self.system_prompt = system_prompt or self.DEFAULT_SYSTEM_PROMPT
        
        # 延迟初始化 LLM 和链
        self._llm: Optional[ChatDeepSeek] = None
        self._chain = None
    
    @property
    def llm(self) -> ChatDeepSeek:
        """懒加载 LLM"""
        if self._llm is None:
            self._llm = ChatDeepSeek(
                model=self.llm_model,
                temperature=self.temperature,
                # 流式输出配置
                # streaming=True,
            )
        return self._llm
    
    @property
    def chain(self):
        """
        获取 RAG 链（懒加载）
        
        链结构：
        Retrieval Chain
            ├── combine_docs_chain
            │   ├── prompt (系统提示词 + 上下文 + 问题)
            │   ├── llm (DeepSeek-chat)
            │   └── output_parser
            └── retriever (向量数据库检索)
        """
        if self._chain is None:
            self._chain = self._build_chain()
        return self._chain
    
    def _build_chain(self):
        """
        构建 RAG 链
        
        使用 create_retrieval_chain 的优势：
        - 自动处理检索和生成的衔接
        - 支持流式输出
        - 便于调试和监控
        """
        # 1. 创建提示词模板
        # {context} 会自动填充检索到的文档
        # {input} 是用户的问题
        prompt = ChatPromptTemplate.from_messages([
            ("system", self.system_prompt),
            ("human", "{input}"),
        ])
        
        # 2. 创建文档处理链（stuff 模式）
        # stuff 模式：将所有检索到的文档塞进一个 prompt
        # 优点：简单快速，适合检索结果不多的情况
        combine_docs_chain = create_stuff_documents_chain(
            llm=self.llm,
            prompt=prompt,
            document_prompt=ChatPromptTemplate.from_template(
                # 自定义每个文档在 prompt 中的格式
                "【文档 {index}】\n内容：{page_content}\n来源：{source}"
            ),
        )
        
        # 3. 创建完整检索链
        retrieval_chain = create_retrieval_chain(
            retriever=self.retriever,
            combine_docs_chain=combine_docs_chain,
        )
        
        return retrieval_chain
    
    def invoke(
        self,
        question: str,
        chat_history: Optional[List[Dict]] = None,
    ) -> Dict[str, Any]:
        """
        执行问答
        
        Args:
            question: 用户问题
            chat_history: 对话历史（用于多轮对话）
            
        Returns:
            {
                "answer": "答案",
                "context": [相关的文档列表],
                "source_documents": [原始 Document 对象]
            }
        """
        logger.info(f"处理问题: {question}")
        
        # 如果有历史记录，构建多轮对话提示
        if chat_history:
            question = self._add_history_to_question(question, chat_history)
        
        # 执行检索和生成
        result = self.chain.invoke({
            "input": question,
        })
        
        # 格式化结果
        return {
            "answer": result["answer"],
            # 返回上下文摘要
            "context": [
                {
                    "content": doc.page_content[:200] + "..." if len(doc.page_content) > 200 else doc.page_content,
                    "source": doc.metadata.get("source", "未知"),
                }
                for doc in result.get("context", [])
            ],
            "source_documents": result.get("context", []),
        }
    
    def _add_history_to_question(
        self,
        question: str,
        chat_history: List[Dict],
    ) -> str:
        """
        将历史对话加入问题
        
        用于多轮对话场景
        """
        history_text = ""
        
        # 只保留最近 N 轮对话
        recent_history = chat_history[-6:]  # 最近 3 轮（6 条消息）
        
        for msg in recent_history:
            role = "用户" if msg["role"] == "user" else "助手"
            history_text += f"{role}：{msg['content']}\n"
        
        if history_text:
            return f"""【对话历史】
{history_text}

【当前问题】
{question}"""
        
        return question
    
    def stream(self, question: str):
        """
        流式执行（用于流式输出）
        
        使用场景：
        - 前端需要实时显示回答
        - 减少等待时间
        """
        # 获取流式响应
        for chunk in self.chain.stream({"input": question}):
            yield chunk


class MultiQueryRAG(RAGChainBuilder):
    """
    多查询 RAG
    
    扩展基础 RAG，增加多查询改写功能：
    1. 从不同角度生成多个查询
    2. 检索更多相关文档
    3. 合并去重后生成答案
    
    适用场景：
    - 用户问题表述不清
    - 需要更全面检索的情况
    """
    
    def __init__(self, *args, num_queries: int = 3, **kwargs):
        super().__init__(*args, **kwargs)
        self.num_queries = num_queries
    
    def _generate_queries(self, question: str) -> List[str]:
        """
        生成多个查询
        
        从不同角度改写问题
        """
        query_prompt = f"""请从不同角度生成 {self.num_queries} 个查询，
用于检索相关文档回答以下问题。

每个查询应该：
- 表达方式不同
- 覆盖不同方面
- 保持核心语义

问题：{question}

请以列表形式输出："""
        
        response = self.llm.invoke(query_prompt)
        
        # 解析响应，提取查询列表
        queries = [q.strip() for q in response.content.split("\n") if q.strip()]
        queries.append(question)  # 保留原问题
        
        return queries[:self.num_queries]
```

```python
class ChatMemory:
    """
    对话记忆管理器
    
    功能：
    1. 存储对话历史
    2. 加载历史对话
    3. 上下文窗口限制
    
    为什么需要记忆管理？
    - LLM 没有持久记忆
    - 每次请求都是独立的
    - 需要手动传入历史对话
    """
    
    def __init__(
        self,
        session_id: str,
        max_messages: int = 20,  # 保留最近 N 条消息
        user_id: Optional[str] = None,
    ):
        """
        初始化记忆管理器
        
        Args:
            session_id: 会话 ID（用于区分不同对话）
            max_messages: 最大保留消息数（超过则截断）
            user_id: 用户 ID（可选）
        """
        self.session_id = session_id
        self.max_messages = max_messages
        self.user_id = user_id
        self._messages: List[dict] = []
    
    def add_user_message(self, content: str):
        """
        添加用户消息
        
        Args:
            content: 消息内容
        """
        self._messages.append({"role": "user", "content": content})
        self._trim_history()
    
    def add_ai_message(self, content: str):
        """
        添加 AI 回复
        
        Args:
            content: 消息内容
        """
        self._messages.append({"role": "assistant", "content": content})
        self._trim_history()
    
    def _trim_history(self):
        """
        截断历史
        
        超过 max_messages 时，保留最近的对话
        """
        if len(self._messages) > self.max_messages:
            # 保留最近的消息
            # 但要保证 user/assistant 交替
            self._messages = self._messages[-self.max_messages:]
    
    def get_messages(self) -> List[dict]:
        """
        获取所有消息
        
        Returns:
            消息列表，每条消息包含 role 和 content
        """
        return self._messages.copy()
    
    def get_recent_messages(self, n: int = 10) -> List[dict]:
        """
        获取最近 N 条消息
        
        Args:
            n: 消息数量
        """
        return self._messages[-n:] if n > 0 else self._messages
    
    def clear(self):
        """清空历史"""
        self._messages = []
    
    def to_langchain_messages(self) -> List[BaseMessage]:
        """
        转换为 LangChain 消息格式
        
        Returns:
            LangChain 消息列表（HumanMessage / AIMessage）
        """
        messages = []
        for msg in self._messages:
            if msg["role"] == "user":
                messages.append(HumanMessage(content=msg["content"]))
            else:
                messages.append(AIMessage(content=msg["content"]))
        return messages
    
    @classmethod
    def from_history(
        cls,
        session_id: str,
        history: List[dict],
        **kwargs,
    ) -> "ChatMemory":
        """
        从历史记录创建记忆
        
        Args:
            session_id: 会话 ID
            history: 历史消息列表
        """
        memory = cls(session_id=session_id, **kwargs)
        memory._messages = history.copy()
        return memory
```

```python
# backend/app/api/document.py
"""
文档管理 API
处理文档上传、解析、删除等请求
"""

from fastapi import APIRouter, UploadFile, File, HTTPException, Depends
from fastapi.responses import JSONResponse
from typing import List
import os
import uuid
import logging

from app.services.document_service import DocumentService, get_document_service
from app.models.schemas import (
    DocumentUploadResponse,
    DocumentListResponse,
    DocumentDeleteResponse,
)
from app.config import get_settings

logger = logging.getLogger(__name__)

router = APIRouter()


@router.post("/upload", response_model=DocumentUploadResponse)
async def upload_document(
    file: UploadFile = File(...),
    document_service: DocumentService = Depends(get_document_service),
):
    """
    上传并处理文档
    
    流程：
    1. 检查文件类型
    2. 保存文件
    3. 解析文档
    4. 分块
    5. 向量化存储
    6. 保存元数据
    
    Args:
        file: 上传的文件
        
    Returns:
        文档 ID 和处理结果
    """
    settings = get_settings()
    
    # 1. 检查文件类型
    if not file.filename:
        raise HTTPException(status_code=400, detail="文件名不能为空")
    
    ext = os.path.splitext(file.filename)[1].lower()
    if ext not in settings.allowed_file_types:
        raise HTTPException(
            status_code=400,
            detail=f"不支持的文件类型: {ext}，支持的类型: {settings.allowed_file_types}",
        )
    
    # 2. 生成唯一文件 ID
    file_id = str(uuid.uuid4())
    filename = f"{file_id}{ext}"
    file_path = os.path.join(settings.upload_dir, filename)
    
    # 确保目录存在
    os.makedirs(settings.upload_dir, exist_ok=True)
    
    try:
        # 3. 保存文件
        content = await file.read()
        
        # 检查文件大小
        if len(content) > settings.max_file_size:
            raise HTTPException(
                status_code=400,
                detail=f"文件过大，最大支持 {settings.max_file_size / 1024 / 1024}MB",
            )
        
        with open(file_path, "wb") as f:
            f.write(content)
        
        logger.info(f"文件保存成功: {file_path}")
        
        # 4. 处理文档
        result = await document_service.process_document(
            file_path=file_path,
            file_id=file_id,
            original_filename=file.filename,
        )
        
        return DocumentUploadResponse(
            success=True,
            document_id=file_id,
            filename=file.filename,
            chunks_count=result.get("chunks_count", 0),
            message="文档上传并处理成功",
        )
        
    except Exception as e:
        logger.error(f"文档处理失败: {e}")
        # 清理已上传的文件
        if os.path.exists(file_path):
            os.remove(file_path)
        raise HTTPException(status_code=500, detail=str(e))


@router.post("/batch-upload")
async def batch_upload_documents(
    files: List[UploadFile] = File(...),
    document_service: DocumentService = Depends(get_document_service),
):
    """
    批量上传文档
    
    Args:
        files: 文件列表
        
    Returns:
        处理结果列表
    """
    results = []
    
    for file in files:
        try:
            # 复用单个上传的逻辑
            response = await upload_document(file, document_service)
            results.append({
                "filename": file.filename,
                "success": True,
                "document_id": response.document_id,
            })
        except Exception as e:
            results.append({
                "filename": file.filename,
                "success": False,
                "error": str(e),
            })
    
    return {"results": results}


@router.get("/list", response_model=DocumentListResponse)
```

```python
@router.post("/chat", response_model=ChatResponse)
async def chat(
    request: ChatRequest,
    chat_service: ChatService = Depends(get_chat_service),
):
    """
    问答接口
    
    Args:
        request: 包含问题和其他参数
        
    Returns:
        回答结果
    """

        result = await chat_service.chat(
            question=request.question,
            session_id=request.session_id,
            use_history=request.use_history,
            history_limit=request.history_limit,
        )
        
        return ChatResponse(
            answer=result["answer"],
            sources=result.get("sources", []),
            context=result.get("context", []),
        )
        
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))


@router.post("/stream")
async def stream_chat(
    request: ChatRequest,
    chat_service: ChatService = Depends(get_chat_service),
):
    """
    流式问答接口
    
    返回 Server-Sent Events (SSE)
    
    Args:
        request: 问答请求
        
    Returns:
        流式响应
    """
    async def event_generator():
        try:
            async for chunk in chat_service.stream_chat(
                question=request.question,
                session_id=request.session_id,
            ):
                # 发送数据
                yield f"data: {chunk}\n\n"
                
        except Exception as e:
            yield f"data: [ERROR] {str(e)}\n\n"
        
        # 结束标记
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
    )

    chat_service: ChatService = Depends(get_chat_service),
):
    """
```

```vue
  })
```
```javascript
  return response.json()

// chatApi 命名导出
export const chatApi = {
  /**
   * 问答
   */
  async chat({ question, session_id, use_history = true, history_limit = 10 }) {
    const response = await fetch(`${API_BASE_URL}/chat/chat`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        question,
        session_id,
        use_history,
        history_limit,
      }),
    })
    return response.json()
  },
  
  /**
   * 流式问答
   */
  streamChat({ question, session_id }, onMessage, onError) {
    const eventSource = new EventSource(`${API_BASE_URL}/chat/stream`)
    
    eventSource.onmessage = (event) => {
      if (event.data === '[DONE]') {
        eventSource.close()
      } else if (event.data.startsWith('[ERROR]')) {
        onError(event.data)
        eventSource.close()
      } else {
        onMessage(event.data)
      }
    }
    
    eventSource.onerror = (error) => {
      onError(error)
      eventSource.close()
    }
    
    return eventSource
  },
}
```
